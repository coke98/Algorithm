N = int(input())
M = int(input())

parent = [i for i in range(N)]
size = [1]*N

def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])
    return parent[x]

def union(x, y):
    rx = find(x)
    ry = find(y)
    
    if rx == ry:
        return
    
    if size[rx] > size[ry]:
        parent[ry] = rx
        size[rx] += size[ry]
    else:
        parent[rx] = ry
        size[ry] += size[rx]

for i in range(N):
    row = input().split()
    for j in range(N):
        if row[j] == '1':
            union(i, j)

path = list(map(int, input().split()))
root = find(path[0]-1)
for city in path:
    if find(city-1) != root:
        print('NO')
        exit()
print('YES')
