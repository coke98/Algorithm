N, K = map(int, input().split())
obj = [0]
for i in range(N):
	W, V = map(int, input().split()) 
	obj.append((W, V))

dp = [[0]*(N+1) for _ in range(K+1)]
for j in range(1, K+1):
	for i in range(1, N+1):
		W, V = obj[i]
		if j - W < 0:
			dp[j][i] = dp[j][i-1]
		else:
			dp[j][i] = max(dp[j][i-1], V + dp[j-W][i-1])
print(dp[K][N])
