### ğŸ’¡ë‚´ ì½”ë“œ

```python
from collections import deque

N, M = map(int, input().split())
sy, sx, ty, tx = map(int, input().split())
sy, sx, ty, tx = sy-1, sx-1, ty-1, tx-1
matrix = [list(input()) for _ in range(N)]

answer = 1
dx = [0, 0, 1, -1]
dy = [1, -1, 0, 0]

found = False

while not found:
    remove = set()
    visited = [[0]*M for _ in range(N)]
    visited[sy][sx] = 1
    q = deque()
    q.append((sx, sy))
    while q:
        x, y = q.popleft()
        for i in range(4):
            nx, ny = x+dx[i], y+dy[i]
            if 0<= nx < M and 0<= ny < N and visited[ny][nx] == 0:
                visited[ny][nx] = 1
                if matrix[ny][nx] == '0':
                    q.append((nx,ny))
                if matrix[ny][nx] == '1':
                    remove.add((nx, ny))
                if matrix[ny][nx] == '#':
                    print(answer)
                    found = True
                    q = deque()
                    break
    for x, y in remove:
        matrix[y][x] = '0'
    answer += 1
```
