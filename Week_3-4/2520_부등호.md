```python
N = int(input())
signs = input().split()

answer = []

visited = [False] * 10

def is_valid(a, b, op):
    return (a < b) if op == '<' else (a > b)

def dfs(depth, path):
    if depth == N + 1:
        answer.append(''.join(map(str, path)))
        return
    for i in range(10):
        if not visited[i]:
            if depth == 0 or is_valid(path[-1], i, signs[depth - 1]):
                visited[i] = True
                dfs(depth + 1, path + [i])
                visited[i] = False

dfs(0, [])

answer.sort()
print(answer[-1])
print(answer[0])
```
