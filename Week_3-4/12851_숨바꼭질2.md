### ðŸ’¡ì½”ë“œ

```python
from collections import deque

N, K = map(int,input().split())

visited = [-1] * 100001

q = deque()
q.append((N,0))
visited[N] = 0

min_time = -1

cnt = 0 

while q:
    X, T = q.popleft()
    if X == K:
        if min_time == -1:
            min_time = T
            cnt += 1
        elif min_time == T:
            cnt += 1
    if not min_time == -1:
        if T+1 > min_time:
            continue
     
    for nx in (2*X, X+1, X-1):
        if 0 <= nx <= 100000:
            if visited[nx] == -1 or visited[nx] == T+1:
                q.append((nx, T+1))
                visited[nx] = T+1
        
print(min_time)
print(cnt)
```
