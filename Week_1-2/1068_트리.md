from collections import deque

N = int(input())
nodes = list(map(int,input().split()))
T = int(input())

def get_leaf(l):
    parent = set()
    for node in l:
        if node == -1:
            continue
        parent.add(node)
    return len(l) - len(parent)

res = []
exc = set()
exc.add(T)
q = deque()
q.append(T)
while q:
    cur = q.popleft()
    for i, node in enumerate(nodes):
        if node == cur:
            q.append(i)
            exc.add(i)
    
for i, node in enumerate(nodes):
    if not i in exc:
        res.append(node)

print(get_leaf(res))
