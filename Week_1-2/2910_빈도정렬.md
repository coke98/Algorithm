```python
from collections import Counter

N, C = map(int,input().split())
l = list(map(int,input().split()))

d = Counter(l)

ans = []

while d:
    m = 0
    mk = 0
    for k in d.keys():
        if d[k] > m:
            m = d[k]
            mk = k
    for i in range(m):
        ans.append(mk)
    del d[mk]
    
print(*ans)
```

```python
from collections import Counter

N, C = map(int,input().split())
l = list(map(int,input().split()))
d = Counter(l).most_common()

ans = []

for num, freq in d:
    ans.extend([num]*freq)

print(*ans)
```
