N, M = map(int, input().split())
dic = {}
for _ in range(M):
    A, B = map(int, input().split())
    if not B in dic:
        dic[B] = [A]
    else:
        dic[B].append(A)
answer = []
max_cnt = 0

def get_num(start, d):
    visited = [False] * (N+1)
    stack = [start]
    visited[start] = True
    cnt = 0

    while stack:
        u = stack.pop()
        for v in d.get(u, []):
            if not visited[v]:
                visited[v] = True
                cnt += 1
                stack.append(v)
    return cnt

for key in dic.keys():
    m = get_num(key, dic)
    if max_cnt == m:
        answer.append(key)
    if max_cnt < m:
        max_cnt = m
        answer = [key]

print(*sorted(answer))
```
