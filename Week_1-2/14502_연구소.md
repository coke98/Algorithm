```python
from itertools import combinations
from collections import deque

N, M = map(int, input().split())

matrix = [list(map(int, input().split())) for _ in range(N)]
blank = []

for y in range(N):
    for x in range(M):
        cur = matrix[y][x]
        if cur == 0:
            blank.append((x,y))
            
            
dx = [0, 0, 1, -1]
dy = [1, -1, 0, 0]

answer = 0

for cbn in combinations(blank, 3):
    visited = [[0]*M for _ in range(N)]
    safe_area = 0
    for y in range(N):
        for x in range(M):
            cur = matrix[y][x]        
            if cur == 0 and (x,y) not in cbn and visited[y][x] == 0:
                q = deque()
                q.append((x,y))
                visited[y][x] = 1
                safe = True
                cnt = 1
                while q:
                    cx, cy = q.popleft()
                    for i in range(4):
                        nx, ny = cx + dx[i], cy + dy[i]
                        if ((nx, ny) not in cbn 
                            and 0 <= nx < M and 0 <= ny < N 
                            and visited[ny][nx] == 0):
                            if matrix[ny][nx] == 0:
                                visited[ny][nx] = 1
                                cnt += 1
                                q.append((nx,ny))
                                
                            if matrix[ny][nx] == 2:
                                safe = False
                if safe:
                    safe_area += cnt
    answer = max(answer, safe_area)
print(answer)

```
