            cur = matrix[y][x]

                q = deque()
                q.append((x,y))
                visited_node.append((x,y))

                while q:
                    x, y = q.popleft()
                    for i in range(4):
                        nx, ny = x+dx[i], y+dy[i]
                        if 0 <= nx < M and 0 <= ny < N and visited[ny][nx] == 0:
                            if matrix[ny][nx] == -1:
                                is_air = True
                            if matrix[ny][nx] == 0:
                                visited[ny][nx] = 1
                                visited_node.append((nx,ny))
                                q.append((nx,ny))
                        
                if is_air:
                    for x, y in visited_node:
                        matrix[y][x] = -1

    # ì¹˜ì¦ˆ ì™¸ê³½ì„  ì°¾ê³  ê³µê¸°ë¡œ ë³€í™˜

                    if 0 <= nx < M and 0 <= ny < N:
                        if matrix[ny][nx] == -1:
                            remove.append((x, y))
                            break
    for x, y in remove:
        matrix[y][x] = -1
